\section{Introducción}

\begin{frame}{Resumen}
\begin{itemize}
\item Identificación experimental del sistema PADP.
\item Modelo promedio a partir de datos válidos.
\item Controladores: PID, ISF-LQR, I-PD y PID discreto.
\item Validación en simulación y experimento.
\item Especificaciones: $t_s \leq 5$ s, $M_p \leq 3\%$.
\end{itemize}
\end{frame}


\section{Metodología}

\begin{frame}{Identificación}
\begin{itemize}
\item 3 conjuntos de datos experimentales.
\item Estimación de $\omega_n$ y $\zeta$.
\item Uso de System Identification Toolbox.
\item Modelo final promedio con aproximación del $87\%$.
\end{itemize}
\end{frame}

\begin{frame}{Controladores}
\begin{itemize}
\item PID continuo (SISOtool).
\item ISF-LQR con acción integral.
\item I-PD por ubicación de polos.
\item PID discreto.
\item Evaluación en simulación y experimento.
\end{itemize}
\end{frame}
