\section{Introducción}          % Sección de introducción

\subsection{Resumen}            % Subsección de resumen
    \begin{frame}{Resumen}
\begin{itemize}
    \item Diseño y validación de un controlador PI para velocidad angular de un motor DC.
    
    \item Identificación del modelo de la planta mediante datos experimentales y análisis estadístico.
    
    \item Diseño del controlador usando lugar de las raíces con restricciones de desempeño:
    \begin{itemize}
        \item $t_s < 300$ ms
        \item $M_p < 3\%$
    \end{itemize}
    
    \item Implementación en tiempo continuo y discreto, con validación en Simulink.
    
    \item Comparación entre simulación y resultados experimentales.
    
    \item Se observaron discrepancias debido a:
    \begin{itemize}
        \item Ruido
        \item Limitaciones del modelo
        \item Efectos no modelados
    \end{itemize}
\end{itemize}
\end{frame}


\subsection{Objetivos Generales}
\begin{frame}{Objetivos Generales}
\begin{itemize}
    \item Diseñar e implementar un controlador PI para regular la velocidad angular de un motor DC, cumpliendo especificaciones de desempeño en términos de tiempo de establecimiento y sobreimpulso.
    \item Analizar el comportamiento del sistema tanto en simulación como en implementación experimental, comparando ambos resultados.
\end{itemize}
\end{frame}

\subsection{Objetivos Específicos}
    \begin{frame}{Objetivos Específicos}
\begin{itemize}
    \item Identificar un modelo matemático del motor DC a partir de datos experimentales mediante análisis estadístico.
    \item Diseñar controladores PI en tiempo continuo utilizando la técnica de lugar de las raíces.
    \item Discretizar el controlador para su implementación digital con un período de muestreo adecuado.
    \item Evaluar el desempeño del sistema mediante simulaciones en Simulink.
    \item Implementar el controlador en una plataforma experimental y analizar su respuesta ante señales de referencia y perturbaciones.
    \item Estudiar el efecto del ruido.
\end{itemize}
\end{frame}   % Subsección de objetivos específicos

\section{Metodología de medición}   % Sección de metodología
    \subsection{Metodología}
        \subsection{Metodología}
\begin{frame}{Metodología}
\begin{itemize}
    \item Etapa 1: Identificación de la planta
    \begin{itemize}
        \item Recolección de múltiples conjuntos de datos experimentales.
        \item Análisis estadístico para seleccionar el dataset más representativo.
        \item Obtención de un modelo de primer orden en MATLAB.
    \end{itemize}
    
    \item Etapa 2: Diseño del controlador
    \begin{itemize}
        \item Diseño de controladores PI en tiempo continuo usando lugar de las raíces.
        \item Cumplimiento de especificaciones ($t_s$, $M_p$).
        \item Discretización con $T_s = 5$ ms.
        \item Validación mediante simulaciones en Simulink.
    \end{itemize}
    
    \item Etapa 3: Implementación y validación
    \begin{itemize}
        \item Implementación en plataforma experimental.
        \item Pruebas con entradas tipo escalón y perturbaciones.
        \item Comparación simulación vs experimental.
        \item Análisis de ruido.
    \end{itemize}
\end{itemize}
\end{frame}     % Subsección de metodología
