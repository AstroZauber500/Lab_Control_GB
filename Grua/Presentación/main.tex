\section{Controladores ISF}

\begin{frame}{Estrategia ISF — Marco Común}
\begin{columns}[T]
  \column{0.50\textwidth}
  \begin{block}{Integral State Feedback (ISF)}
  \begin{itemize}
    \item Estado integral asociado al error:\\
          $\dot{x}_I = r - C_r x$
    \item Ley de control:\\
          $u = -Kx + K_I\,x_I$
    \item Sistema \textbf{aumentado} (5 estados).
    \item $\text{rank}(\hat{\mathcal{M}}_c) = 5$ \good{$\checkmark$}
  \end{itemize}
  \end{block}

  \begin{exampleblock}{Beneficios}
  \scriptsize
  \begin{itemize}
    \item $e_{ss}=0$ garantizado ante escalón.
    \item Polos de lazo cerrado asignables libremente.
    \item Atenuación de oscilación del péndulo.
  \end{itemize}
  \end{exampleblock}

  \column{0.46\textwidth}

\begin{frame}{Pole Placement — Diseño}
\begin{columns}[T]
  \column{0.50\textwidth}
  \begin{block}{Metodología}
  \begin{enumerate}
    \item Construir sistema \textbf{aumentado} (4+1 estados).
    \item Seleccionar \textbf{5 polos deseados} en $\mathbb{C}^-$.
    \item Calcular $\hat{K}=[K\;|\;K_I]$ con \texttt{place()} en MATLAB.

  \end{enumerate}
  \end{block}

\subsection{LQR}

\begin{frame}{LQR — Regla de Bryson}
\begin{columns}[T]
  \column{0.50\textwidth}
  \begin{block}{Criterio de diseño}
  \centering\small
  \[
    J = \int_0^\infty \!\bigl(x^TQx + u^TRu\bigr)\,dt
  \]
  \[
    Q_{ii} = \frac{1}{x_{i,\max}^2}, \qquad
    R = \frac{1}{u_{\max}^2}
  \]
  \end{block}

  \vspace{4pt}
  \begin{block}{Ganancias obtenidas}
  \centering\scriptsize
  \[
    \hat{K}_{LQR} = \begin{bmatrix}
      80.04 & {-}23.19 & 7.13 & {-}7.01 & {-}40.00
    \end{bmatrix}
  \]
  \[
    K = \begin{bmatrix}80.04 & {-}23.19 & 7.13 & {-}7.01\end{bmatrix}
  \]
  \[
    K_I = -40.0
  \]
  \textbf{Polos LC:}
  $-10.47,\;-1.50\pm 6.12j,\;-0.79\pm 0.49j$
  \end{block}

  \column{0.46\textwidth}
  \centering\scriptsize
  \rowcolors{2}{rowsim}{white}
  \begin{tabular}{p{1.5cm}ccc}
    \rowcolor{rowhdr}
    \color{white}\textbf{Estado} &
    \color{white}$x_{\max}$ &
    \color{white}$Q_{ii}$ \\
    Posición $p$         & 0.112 m       & \textbf{80.0} \\
    Ángulo $\theta$      & 0.209 rad     & \textbf{23.0} \\
    Vel.\ lineal $\dot{p}$    & 1.0 m/s  & 1.0 \\
    Vel.\ ang.\ $\dot\theta$ & 0.354 r/s & \textbf{8.0} \\

  \begin{exampleblock}{\scriptsize Peso dominante}

  $Q_{\theta\theta}=23$ atenúa oscilación del péndulo.
  \end{exampleblock}
\end{columns}
\end{frame}

\section{Conclusiones}
%=============================================================================

\begin{frame}{Conclusiones}
\begin{enumerate}
  \item Se identificó la planta grúa obteniendo un modelo SIMO 4.º orden con\\
        \good{CV\,$<\,$2\,\%} en todos los parámetros — modelo confiable.

  \item El controlador \textbf{LQR} cumplió las especificaciones:
        \[
          t_s^{\text{sim}}=4.70\,\text{s},\;
          M_p^{\text{sim}}=0.60\,\%,\;
          t_s^{\text{exp}}=5.25\,\text{s},\;
          M_p^{\text{exp}}=0\,\%,\;
          e_{ss}=0
        \]

  \item La \textbf{acción integral} del esquema ISF garantizó error estacionario nulo.

  \item Diferencias sim.\ vs.\ exp.\ se atribuyen a:
        fricción $\cdot$ zona muerta $\cdot$ saturación del actuador.

  \item La ausencia de sobreimpulso experimental indica mayor amortiguamiento
        físico que el del modelo linealizado.
\end{enumerate}
\end{frame}

\subsection{Recomendaciones}

\begin{frame}{Recomendaciones}
\begin{columns}[T]
  \column{0.50\textwidth}
  \begin{block}{Identificación}
  \begin{itemize}
    \item Incorporar más sets experimentales para reducir CV.
    \item Aplicar criterios estadísticos formales (IQR, Chauvenet) para descartar atípicos.
  \end{itemize}
  \end{block}

  \begin{block}{Modelado}
  \begin{itemize}
    \item Incluir efectos no lineales:\\
          fricción $\cdot$ zona muerta $\cdot$ saturación.
  \end{itemize}
  \end{block}

  \column{0.46\textwidth}
  \begin{block}{Control}
  \begin{itemize}
    \item Ajustar pesos de Bryson conservadoramente para reducir esfuerzo de control.
    \item Explorar técnicas robustas ($H_\infty$, LQG) para mejor atenuación del péndulo.
    \item Agregar realimentación de $\theta$ como salida secundaria para control explícito del ángulo.
  \end{itemize}
  \end{block}
\end{columns}
\end{frame}
